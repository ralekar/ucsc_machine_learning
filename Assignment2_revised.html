def get_pca(train):
    X = np.vstack(train)
    u = get_mean(X)
    Z = X - u
    c = get_covarience(train)
    v = get_eigen_values(c)
    return np.dot(Z,v.transpose())

def get_covarience(train) :
    X = np.vstack(train)
    u = get_mean(X)
    Z = X - u
    return np.cov(Z.transpose())

def get_eigen_values(c):
    w,v = np.linalg.eig(c)
    return  -1 * v.transpose()

def plot_scatter(P,length_first,number_of_components):
    x1,y1,x2,y2 = [],[],[],[]
    for i in range(0,length_first):
        a= np.reshape((P[i].real)[:number_of_components],(1,2))
        x1.append(a[0][0])
        y1.append(a[0][1])
    for i in range(length_first,len(P)):
        a= np.reshape((P[i].real)[:number_of_components],(1,2))
        x2.append(a[0][0])
        y2.append(a[0][1])
    return (x1,y1),(x2,y2)

class classifier :
    
    def __init__(self,p1,p2):
        self.p11 = np.asarray(map(lambda z : z,p1[0]))
        self.p12 = np.asarray(map(lambda z : z,p1[1]))
        self.p21 = np.asarray(map(lambda z : z,p2[0]))
        self.p22 = np.asarray(map(lambda z : z,p2[1]))
        self.M = 25
        self.p1_bins = {}
        self.p2_bins = {}
        self.util()
        
    def util(self):
        for i in range(len(self.p11)):
            self.set_bins_p1((self.p11[i],self.p12[i]))
        for i in range(len(self.p21)):
            self.set_bins_p2((self.p21[i],self.p22[i]))    
            
    def get_rc_p1(self,v):
        row = np.round(1+((self.M-1)*((v[0]-np.min(self.p11))/(np.max(self.p11)-np.min(self.p11)))))
        col = np.round(1+((self.M-1)*((v[1]-np.min(self.p12))/(np.max(self.p12)-np.min(self.p12)))))
        
        return (row,col)
    
    def get_rc_p2(self,v):
        row = np.round(1+((self.M-1)*((v[0]-np.min(self.p21))/(np.max(self.p21)-np.min(self.p21)))))
        col = np.round(1+((self.M-1)*((v[1]-np.min(self.p22))/(np.max(self.p22)-np.min(self.p22)))))
        
        return (row,col)
    
    def set_bins_p1(self,v):
        r,c = self.get_rc_p1(v)
        if (r,c) in self.p1_bins:
            self.p1_bins[(r,c)]+=1
        else:
            self.p1_bins[(r,c)] = 1
    
    def set_bins_p2(self,v):
        r,c = self.get_rc_p2(v)
        if (r,c) in self.p2_bins:
            self.p2_bins[(r,c)]+=1
        else:
            self.p2_bins[(r,c)] = 1
    
    def hist(self,v):
        
        pcount1 = 0.0
        pcount2 = 0.0
        
        if self.get_rc_p1(v) in self.p1_bins:
            pcount1 = self.p1_bins[self.get_rc_p1(v)]
        if self.get_rc_p2(v) in self.p2_bins:
            pcount2 = self.p2_bins[self.get_rc_p2(v)]
        
        if pcount1 == 0.0 and pcount2 == 0.0 : 
           return "Indeterminate",0.0 
        if pcount1 == pcount2:
           return "Indeterminate",0.5  
        if pcount1 > pcount2:
           return "5",(pcount1/(pcount2+pcount1))
        if pcount1 < pcount2:
           return "6",(pcount2/(pcount2+pcount1))
        
        return "Indeterminate",0.0
